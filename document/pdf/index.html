      Dialog.alert({
        title: "温馨提示",

    }
    const reader = new FileReader();
    reader.readAsDataURL(file);
    reader.onload = (evt) => {
      var imgNum = evt.target.result.split(";base64,")[1]; // 接口要的参数， 视频的base64
    }
 }
